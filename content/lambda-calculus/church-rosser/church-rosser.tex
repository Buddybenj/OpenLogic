\chapter{The Church-Rosser Property}

\begin{defn}[Church-Rosser property, CR]
  A relation $R$ on terms is said to satisfy \emph{Church-Rosser
    property} iff:
  \begin{center}
    If $R P M$ and $R P N$, then there exsits some $Q$ such that $R M
    Q$ and $R N Q$.
  \end{center}
\end{defn}
